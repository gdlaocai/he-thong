// ========== HỆ THỐNG ĐÁNH GIÁ XẾP LOẠI NHÂN VIÊN ========== // Cập nhật: Tích hợp Động cơ thẩm định quy tắc động 7 cột (Đồng bộ cấu trúc GV) function f_x_m_chinh(dataRows) { let diemP1 = 0, diemP2 = 0; let diemThangGoc = 0; let isXSThang = false, isTotThang = false, isHTThang = false, isCHTThang = false; // 1. QUÉT VÀ TÍNH ĐIỂM CƠ BẢN dataRows.forEach(r => { let v_str = r.giaTri || ""; let v_num = r.so || 0; // Xử lý điểm từ thẻ Select if (r.isSelect) diemP2 += r.selectDataP; // Xử lý hệ số từ Quy chế (Cột C) if (r.quyChe) { let tgClean = r.quyChe.replace(/\s/g, ''); let heSoMatch = tgClean.match(/^\(([\+\-]?\d+(\.\d+)?)\)$/); if (heSoMatch) { let heSo = parseFloat(heSoMatch[1]); if (!isNaN(heSo)) { diemP2 += v_num * heSo; } } } // Xử lý hệ số quy đổi từ Data-qd if (r.q_d && !(r.quyChe && r.quyChe.match(/^\([\+\-]?\d+(\.\d+)?\)$/))) { let qd = r.q_d.replace(/\s/g, ''); if (qd.startsWith('(+') || qd.startsWith('(-')) { let heSo = parseFloat(qd.match(/[\+\-]\d+(\.\d+)?/)[0]); if (!isNaN(heSo)) diemP2 += (v_num * heSo); } } // Bóc tách điểm thành tích if (r.ten.includes("thành tích") || r.quyChe.includes("Quy chế")) { let m_d_1 = v_str.match(/=\s*([\d\.]+)\s*điểm/g); if (m_d_1) m_d_1.forEach(m => diemP2 += parseFloat(m.replace(/[^\d\.]/g, '')) || 0); } // Xếp loại hàng tháng (Phần 1) if (r.isDongBo || r.ten.includes("tháng")) { diemThangGoc = v_num; if (v_str.includes("xuất sắc")) { isXSThang = true; isHTThang = true; diemP1 = 100 + diemThangGoc; } else if (v_str.includes("tốt")) { isTotThang = true; isHTThang = true; diemP1 = 80 + diemThangGoc; } else if (v_str.includes("chưa hoàn thành") || v_str.includes("cht")) { isCHTThang = true; diemP1 = 0 + diemThangGoc; } else if (v_str.includes("hoàn thành") || v_str.includes("ht")) { isHTThang = true; diemP1 = 60 + diemThangGoc; } else { diemP1 = diemThangGoc; } } }); // 2. THẨM ĐỊNH QUY TẮC ĐỘNG 7 CỘT (Vùng quét và tham chiếu chéo) let datXS = true, datTot = true, datHT = true; const evalPrizeCondition = (cond, valStr, row) => { if (!row.ten.includes("thành tích") && !row.quyChe.includes("Quy chế")) return null; const lines = valStr.split('\n'); let bestScore = 0; const prizes = []; lines.forEach(line => { let g = 0, c = 0; if(line.includes("nhất")) g = 4; else if(line.includes("nhì")) g = 3; else if(line.includes("ba")) g = 2; else if(line.includes("kk")) g = 1; if(line.includes("quốc gia")) c = 3; else if(line.includes("tỉnh")) c = 2; else if(line.includes("xã") || line.includes("cụm")) c = 1; if (g > 0 && c > 0) { let score = c * 10 + g; prizes.push({score, cap: c, hang: g}); if(score > bestScore) bestScore = score; } }); cond = cond.trim(); let op = ""; let rest = cond; const opMatch = cond.match(/^(>=|<=|>|<|==)\s*/); if (opMatch) { op = opMatch[1]; rest = cond.substring(opMatch[0].length).trim(); } let cap = 0, hang = 0; const lowerRest = rest.toLowerCase(); if (lowerRest.includes("quốc gia")) cap = 3; else if (lowerRest.includes("tỉnh")) cap = 2; else if (lowerRest.includes("xã") || lowerRest.includes("cụm")) cap = 1; if (lowerRest.includes("nhất")) hang = 4; else if (lowerRest.includes("nhì")) hang = 3; else if (lowerRest.includes("ba") || lowerRest.includes("giải 3")) hang = 2; else if (lowerRest.includes("kk") || lowerRest.includes("khuyến khích")) hang = 1; if (cap === 0 || hang === 0) return null; const requiredScore = cap * 10 + hang; if (op === "") { return prizes.some(p => p.cap === cap && p.hang === hang); } else { switch (op) { case ">=": return bestScore >= requiredScore; case ">": return bestScore > requiredScore; case "<=": return bestScore <= requiredScore; case "<": return bestScore < requiredScore; case "==": return bestScore === requiredScore; default: return false; } } }; const evalCondition = (cond, valNum, valStr, row) => { cond = cond.trim().toLowerCase(); if (cond === "" || cond === "any") return true; const prizeResult = evalPrizeCondition(cond, valStr, row); if (prizeResult !== null) return prizeResult; let isNumericCond = cond.match(/^[><=]+[\d\.]+$/); if (isNumericCond) { if (cond.startsWith(">=")) return valNum >= parseFloat(cond.replace(">=", "")); if (cond.startsWith("<=")) return valNum <= parseFloat(cond.replace("<=", "")); if (cond.startsWith(">")) return valNum > parseFloat(cond.replace(">", "")); if (cond.startsWith("<")) return valNum < parseFloat(cond.replace("<", "")); if (cond.startsWith("==")) return valNum === parseFloat(cond.replace("==", "")); } if (cond.match(/^[\d\.]+$/)) return valNum === parseFloat(cond); if (cond.startsWith(">=") && isNaN(cond.replace(">=", ""))) { let hierarchy = [ ["yếu", "kém", "chưa đạt", "không có", "chưa hoàn thành"], ["trung bình", "cấp trường", "hoàn thành"], ["khá", "cấp xã", "cấp cụm", "hoàn thành tốt"], ["tốt", "giỏi", "cấp tỉnh", "cấp huyện", "xuất sắc"], ["cấp quốc gia"] ]; let targetText = cond.replace(">=", "").trim(); let targetLevel = -1, currentLevel = -1; hierarchy.forEach((arr, idx) => { if(arr.some(w => targetText.includes(w))) targetLevel = idx; }); hierarchy.forEach((arr, idx) => { if(arr.some(w => valStr.includes(w))) currentLevel = idx; }); if (targetLevel !== -1 && currentLevel !== -1) return currentLevel >= targetLevel; } if (row.ten.includes("thành tích") || row.quyChe.includes("Quy chế")) { let lines = valStr.split('\n'); let bestScore = 0, countTinh = 0; lines.forEach(line => { let g = 0, c = 0; if(line.includes("nhất")) g = 4; else if(line.includes("nhì")) g = 3; else if(line.includes("ba")) g = 2; else if(line.includes("kk")) g = 1; if(line.includes("quốc gia")) c = 3; else if(line.includes("tỉnh")) { c = 2; countTinh += (parseInt(line.match(/\+\s*(\d+)/)?.[1]) || 1); } else if(line.includes("xã") || line.includes("cụm")) c = 1; let score = c * 10 + g; if(score > bestScore) bestScore = score; }); cond = cond.replace(/\s+/g, ""); if (cond.includes("tỉnh") && cond.match(/>=(\d+)/)) { let req = parseInt(cond.match(/>=(\d+)/)[1]); return countTinh >= req || bestScore >= 30; } let targetScore = 0; if (cond.includes("nhấtxã") || cond.includes("nhấtcụm")) targetScore = 14; else if (cond.includes("nhìcụm") || cond.includes("nhìxã")) targetScore = 13; else if (cond.includes("nhấttỉnh")) targetScore = 24; if (cond.includes(">=")) return bestScore >= targetScore; return bestScore === targetScore; } return valStr.includes(cond); }; const evalCell = (cellFormula, valNum, valStr, row) => { if (!cellFormula) return true; let parts = cellFormula.split("||"); for (let p of parts) { p = p.trim().toLowerCase(); if (row.ten.includes("kết quả xếp loại hàng tháng")) { if (p === "xuất sắc" && isXSThang) return true; if (p === "hoàn thành tốt" && isTotThang) return true; if (p === "hoàn thành" && isHTThang) return true; if (p.includes(">") && diemThangGoc > parseFloat(p.replace(">",""))) return true; } else { if (evalCondition(p, valNum, valStr, row)) return true; } } return false; }; // Đánh giá từng hàng cấu hình dataRows.forEach(r => { if (r.ten === "" || r.ten.includes("phần 1") || r.ten.includes("phần 2")) return; if (r.g_xs && !evalCell(r.g_xs, r.so, r.giaTri, r)) datXS = false; if (r.g_htt && !evalCell(r.g_htt, r.so, r.giaTri, r)) datTot = false; if (r.g_ht && !evalCell(r.g_ht, r.so, r.giaTri, r)) datHT = false; }); if (isCHTThang) { datXS = false; datTot = false; datHT = false; } // 3. TỔNG HỢP VÀ KẾT LUẬN let k_t = "CHƯA HOÀN THÀNH NHIỆM VỤ"; let k_m = "#dc3545"; let k_t_t = diemP1 + diemP2; // Kiểm tra xem sheet CHITIEUNV có cấu hình cột quy tắc không let hasDynamicRules = dataRows.some(r => r.g_xs || r.g_htt || r.g_ht); if (hasDynamicRules) { // Áp dụng Động cơ thẩm định quy tắc 7 cột if (datXS) { k_t = "HOÀN THÀNH XUẤT SẮC NHIỆM VỤ"; k_m = "#28a745"; } else if (datTot) { k_t = "HOÀN THÀNH TỐT NHIỆM VỤ"; k_m = "#007bff"; } else if (datHT) { k_t = "HOÀN THÀNH NHIỆM VỤ"; k_m = "#fd7e14"; } } else { // Dự phòng: Xếp loại theo thang điểm cấu hình cũ (nếu CHITIEUNV chưa điền) if (k_t_t >= 90) { k_t = "HOÀN THÀNH XUẤT SẮC NHIỆM VỤ"; k_m = "#28a745"; } else if (k_t_t >= 70) { k_t = "HOÀN THÀNH TỐT NHIỆM VỤ"; k_m = "#007bff"; } else if (k_t_t >= 50) { k_t = "HOÀN THÀNH NHIỆM VỤ"; k_m = "#fd7e14"; } } return { diemP1: diemP1 % 1 === 0 ? diemP1 : diemP1.toFixed(1), diemP2: diemP2 % 1 === 0 ? diemP2 : diemP2.toFixed(1), tongDiem: k_t_t % 1 === 0 ? k_t_t : k_t_t.toFixed(1), k_t: k_t, k_m: k_m }; }
