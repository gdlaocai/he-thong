} catch (loie) { return ContentService.createTextOutput(JSON.stringify({ tCong: false, tBao: "Lỗi Server: " + loie.message })).setMimeType(ContentService.MimeType.JSON); } } function moBangAnt(idb, ten) { if (!idb) throw new Error("Chưa có ID cho file " + ten); try { return SpreadsheetApp.openById(idb.trim()); } catch (loie) { throw new Error("Lỗi mở file '" + ten + "'."); } } function damBaoCot(trang, socotyc) { var cotmax = trang.getMaxColumns(); if (cotmax < socotyc) trang.insertColumnsAfter(cotmax, socotyc - cotmax); } function dangNhapHt(gMai) { var taiKhoanTruyCap = String(gMai).trim().toLowerCase(); if (!taiKhoanTruyCap) return { tCong: false, tBao: "Lỗi: Không nhận định dạng được tài khoản từ hệ thống." }; const btap = moBangAnt(idKbao, "ThietLap"); const trang = btap.getSheetByName("ThietLap"); if (!trang) return { tCong: false, tBao: "Lỗi: Không tìm thấy dữ liệu cấu hình tại ThietLap." }; const dcuoi = trang.getLastRow(); if (dcuoi < 2) return { tCong: false, tBao: "Hệ thống chưa có dữ liệu danh sách tài khoản." }; const dlb = trang.getRange(2, 1, dcuoi - 1, 3).getDisplayValues(); var ndtim = null; for (var i = 0; i < dlb.length; i++) { var cotA_TaiKhoan = String(dlb[i][0]).trim().toLowerCase(); if (cotA_TaiKhoan === "") continue; } else if (cotC_Quyen === 'x') { vTro = 'ngdung'; } ndtim = { tenn: cotB_Ten ? cotB_Ten : taiKhoanTruyCap, vTro: vTro, tKhoan: taiKhoanTruyCap }; break; } } if (ndtim) return { tCong: true, dLieu: ndtim }; } var bns = moBangAnt(idNsu, "NHANSU"); var tns = bns.getSheetByName("NHANSU"); damBaoCot(tns, 34); var dcns = tns.getLastRow(); if (dcns >= 5) dsnsu = tns.getRange(5, 1, dcns - 4, 34).getDisplayValues(); if (locNam && locNam !== "") dsnsu = dsnsu.filter(function(dong) { return dong[0] == locNam; }); return { chinh: dlchinh, nsu: dsnsu }; } function chuanHoaNgay(d) { if (!d) return ""; var s = String(d).trim(); var parts = s.split('/'); if (parts.length === 3) { var day = parseInt(parts[0], 10); var month = parseInt(parts[1], 10); var year = parts[2].trim(); return (day < 10 ? "0" + day : day) + '/' + (month < 10 ? "0" + month : month) + '/' + year; } return s; } function timDong(dsdl, nam, mso, ngaysinh) { var nsNorm = chuanHoaNgay(ngaysinh); for (var i = 0; i < dsdl.length; i++) { var isNam = String(dsdl[i][0]).trim() === String(nam).trim(); var isMso = String(dsdl[i][1]).trim() === String(mso).trim(); if (isNam && isMso) { var isDate2 = chuanHoaNgay(dsdl[i][2]) === nsNorm; var isDate3 = chuanHoaNgay(dsdl[i][3]) === nsNorm; if (isDate2 || isDate3) return i; const inNgaySinh = (dlForm[36] !== undefined && dlForm[36] !== "") ? dlForm[36] : dlForm[2]; const dcns = tns.getLastRow(); var dlsan = []; if (dcns >= 5) dlsan = tns.getRange(5, 1, dcns - 4, 4).getDisplayValues(); } var dongt = cso + 5; tns.getRange(dongt, 1, 1, 17).setValues([dlghi]); return { tCong: true, tBao: "Đã cập nhật dữ liệu năm " + dlForm[0] + "!" }; } else { if (vTro === 'ngdung') return { tCong: false, tBao: "Bạn không có quyền thêm mới hồ sơ!" }; var dongmoi = Math.max(tns.getLastRow() + 1, 5); tns.getRange(dongmoi, 1, 1, 17).setValues([dlghi]); return { tCong: true, tBao: "Đã thêm mới hồ sơ năm " + dlForm[0] + "!" }; } } function luuNhieuNsu(mangDl, tKhoan, vTro, tenn) { if (vTro !== 'qtriv' && vTro !== 'ngdung') return { tCong: false, tBao: "Không có quyền thực hiện!" }; const bns = moBangAnt(idNsu, "NHANSU"); const tns = bns.getSheetByName("NHANSU"); damBaoCot(tns, 34); if (!mangDl || mangDl.length === 0) return { tCong: true, tBao: "Không có dữ liệu." }; const dcns = tns.getLastRow(); var dlghi = dlForm.slice(0, 17); if (cso !== -1) { if (vTro === 'ngdung' && searchNam !== dlForm[0]) { loiXayRa = true; return; } var dongt = cso + 5; tns.getRange(dongt, 1, 1, 17).setValues([dlghi]); } else { if (vTro === 'ngdung') { loiXayRa = true; return; } var dcmoi = tns.getLastRow(); var dongmoi = Math.max(dcmoi + 1, 5); tns.getRange(dongmoi, 1, 1, 17).setValues([dlghi]); dlsan.push([dlForm[0], dlForm[1], dlForm[2], ""]);
